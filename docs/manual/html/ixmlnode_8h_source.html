29bool XValue(const std::string &value) {
30 std::string v(value);
31 util::string::Trim(v);

33 || util::string::IEquals(v, "ON")
34 || util::string::IEquals(v, "T", 1) // True/False
35 || util::string::IEquals(v, "Y", 1); // Yes/No
